"use client";
import { useState, useEffect } from "react";
import "../../public/Slides.css";

const noticias = [
  {
    titulo: "Actualización de normativa vigente",
    imagen: "/Img-Slides/worldNews.png",
    descripcion: "Revisión completa de los protocolos actuales"
  },
  {
    titulo: "Nueva designación de fiscales",
    imagen: "/Img-Slides/investigacion.jpg",
    descripcion: "Nombramientos oficiales publicados"
  },
  {
    titulo: "Renovación tecnológica en curso",
    imagen: "/Img-Slides/innovacionTec.jpg",
    descripcion: "Modernización de sistemas internos"
  },
  {
    titulo: "Capacitación al personal",
    imagen: "/Img-Slides/incendioNews.png",
    descripcion: "Programas de formación continua"
  },
];

export default function Slides() {
  const [currentSlide, setCurrentSlide] = useState(0);

  useEffect(() => {
    const interval = setInterval(() => {
      setCurrentSlide((prev) => (prev + 1) % noticias.length);
    }, 3000); // Cambia cada 3 segundos

    return () => clearInterval(interval);
  }, []);

  const nextSlide = () => {
    setCurrentSlide((prev) => (prev + 1) % noticias.length);
  };

  const prevSlide = () => {
    setCurrentSlide((prev) => (prev - 1 + noticias.length) % noticias.length);
  };

  const goToSlide = (index) => {
    setCurrentSlide(index);
  };

  return (
    <div className="minimal-carousel">
      <div className="carousel-track">
        {noticias.map((noticia, index) => (
          <div
            key={index}
            className={`carousel-item ${index === currentSlide ? 'active' : ''}`}
          >
            <div className="item-content">
              <div className="text-content">
                <h3 className="item-title">{noticia.titulo}</h3>
                <p className="item-description">{noticia.descripcion}</p>
              </div>
              
              <div className="image-content">
                <img 
                  src={noticia.imagen} 
                  alt={noticia.titulo}
                  className="item-image"
                  onError={(e) => {
                    e.target.style.display = 'none';
                    e.target.nextSibling.style.display = 'flex';
                  }}
                />
                <div className="image-fallback">
                  <span>📋</span>
                </div>
              </div>
            </div>
          </div>
        ))}
      </div>

      {/* Controles */}
      <div className="carousel-controls">
        <button className="control-btn" onClick={prevSlide}>
          ←
        </button>
        
        <div className="indicators">
          {noticias.map((_, index) => (
            <button
              key={index}
              className={`indicator ${index === currentSlide ? 'active' : ''}`}
              onClick={() => goToSlide(index)}
            />
          ))}
        </div>
        
        <button className="control-btn" onClick={nextSlide}>
          →
        </button>
      </div>
    </div>
  );
}
